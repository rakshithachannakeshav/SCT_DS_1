import pandas as pd
import matplotlib.pyplot as plt

# 1. Load CSV (skip metadata rows)
df = pd.read_csv(r"API_SP.POP.TOTL_DS2_en_csv_v2_571486.csv", skiprows=4)

# 2. Keep only India + Population total
india = df[(df["Country Name"] == "India") & (df["Indicator Name"] == "Population, total")]

# 3. Drop non-year columns (Country, Code, Indicator, and trailing NaN column)
india = india.drop(columns=["Country Name","Country Code","Indicator Name","Indicator Code","Unnamed: 69"])

# 4. Melt year columns into rows
india_years = india.melt(var_name="Year", value_name="Population")

# 5. Clean Year → convert to integer
india_years["Year"] = india_years["Year"].astype(int)

# 6. Plot India’s population trend
plt.figure(figsize=(10,6))
plt.plot(india_years["Year"], india_years["Population"], marker="o", color="teal")
plt.title("India - Population, total (1960–2024)")
plt.xlabel("Year")
plt.ylabel("Population")
plt.grid(True)
plt.show()

countries = ["India", "China", "United States"]

subset = subset.melt(id_vars="Country Name", var_name="Year", value_name="Population")
subset["Year"] = subset["Year"].astype(int)

plt.figure(figsize=(12,6))
for country in countries:

from sklearn.linear_model import LinearRegression
import numpy as np

X = india_years["Year"].values.reshape(-1,1)
y = india_years["Population"].values

model = LinearRegression()
model.fit(X, y)

future_years = np.arange(2025, 2036).reshape(-1,1)
future_pop = model.predict(future_years)

plt.figure(figsize=(10,6))
plt.plot(india_years["Year"], india_years["Population"], label="Historical", marker="o")
plt.plot(future_years, future_pop, label="Forecast", marker="x", linestyle="--", color="red")
plt.title("India Population Forecast (2025–2035)")
plt.xlabel("Year")
plt.ylabel("Population")
plt.legend()
plt.grid(True)
plt.show()
